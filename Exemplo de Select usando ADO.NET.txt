
            var query = "SELECT Id, Nome FROM Contato";
            using SqlConnection conn = new("Data Source=./;Initial Catalog=Agendinha;Integrated Security=True;TrustServerCertificate=True");
            using SqlCommand cmd = new(query, conn);
            conn.Open();
            var reader = cmd.ExecuteReader();

            List<Contato> contatos = new();
            if (reader.HasRows)
            {
                while (reader.Read())
                {
                    Contato contato = new()
                    {
                        Id = Convert.ToInt32(reader[nameof(Contato.Id)]),
                        Nome = reader[nameof(Contato.Nome)].ToString()
                        
                    };

                    contatos.Add(contato);
                }
            }
